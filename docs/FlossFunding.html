def base_words(num_valid_words)
  return [] if num_valid_words.nil? || num_valid_words.zero?
  File.open(::FlossFunding::BASE_WORDS_PATH, "r") do |file|
    lines = []
    num_valid_words.times { lines << file.gets.chomp }
    lines
  end
end
